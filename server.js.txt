import express from 'express';
import crypto from 'crypto';
import dotenv from 'dotenv';

dotenv.config();

const app = express();
const PORT = process.env.PORT || 3000;

// Environment Variables
const ASANA_PAT = process.env.ASANA_PAT;
const USPTO_API_KEY = process.env.USPTO_API_KEY;

// Asana Custom Field GIDs (Replace these with your actual Custom Field GIDs in Asana)
const CF_APP_NUMBER = process.env.CF_APP_NUMBER_GID || "YOUR_APP_NUMBER_FIELD_GID";
const CF_PATENT_TITLE = process.env.CF_PATENT_TITLE_GID || "YOUR_TITLE_FIELD_GID";
const CF_STATUS = process.env.CF_STATUS_GID || "YOUR_STATUS_FIELD_GID";

// We need the raw body to verify the Asana webhook HMAC signature
app.use(express.json({
    verify: (req, res, buf) => {
        req.rawBody = buf;
    }
}));

// Global variable to store the webhook secret provided during the Asana handshake
let webhookSecret = process.env.ASANA_WEBHOOK_SECRET || "";

/**
 * USPTO API Call
 * Fetches patent data from the USPTO Open Data Portal based on application number.
 */
async function fetchUSPTOData(applicationNumber) {
    console.log(`Fetching USPTO data for application: ${applicationNumber}`);
    
    try {
        // NOTE: USPTO API syntax may vary based on exact endpoint used (PEDS vs Open Data). 
        // This targets the modern Open Data Portal API.
        const url = `https://api.uspto.gov/api/v1/patent/applications/search`;
        
        // Formulate search for the specific application number
        const requestBody = {
            "searchText": `applicationNumberText:${applicationNumber}`,
            "pagination": { "limit": 1, "offset": 0 }
        };

        const response = await fetch(url, {
            method: 'POST',
            headers: {
                'accept': 'application/json',
                'Content-Type': 'application/json',
                'X-API-KEY': USPTO_API_KEY
            },
            body: JSON.stringify(requestBody)
        });

        if (!response.ok) {
            throw new Error(`USPTO API error: ${response.status} ${response.statusText}`);
        }

        const data = await response.json();
        
        // Extract relevant data (Adjust field names based on exact USPTO response structure)
        if (data && data.results && data.results.length > 0) {
            const patentRecord = data.results[0];
            return {
                title: patentRecord.inventionTitle || "N/A",
                status: patentRecord.applicationStatusDescriptionText || "Unknown",
                patentNumber: patentRecord.patentNumber || "Pending"
            };
        }
        return null;
    } catch (error) {
        console.error("Error fetching from USPTO:", error);
        return null;
    }
}

/**
 * Asana API Call
 * Updates the custom fields on an Asana task.
 */
async function updateAsanaTask(taskGid, usptoData) {
    console.log(`Updating Asana task ${taskGid} with data:`, usptoData);
    
    const url = `https://app.asana.com/api/1.0/tasks/${taskGid}`;
    
    const requestBody = {
        data: {
            custom_fields: {
                [CF_PATENT_TITLE]: usptoData.title,
                [CF_STATUS]: usptoData.status
            }
        }
    };

    try {
        const response = await fetch(url, {
            method: 'PUT',
            headers: {
                'accept': 'application/json',
                'content-type': 'application/json',
                'authorization': `Bearer ${ASANA_PAT}`
            },
            body: JSON.stringify(requestBody)
        });

        if (!response.ok) {
            const errBody = await response.text();
            throw new Error(`Asana API error: ${response.status} - ${errBody}`);
        }
        
        console.log(`Successfully updated Asana task ${taskGid}.`);
    } catch (error) {
        console.error("Error updating Asana:", error);
    }
}

/**
 * Process individual Asana events
 */
async function processTaskEvent(taskGid) {
    try {
        // 1. Fetch the full task details to get the current Custom Field values
        const response = await fetch(`https://app.asana.com/api/1.0/tasks/${taskGid}`, {
            headers: { 'authorization': `Bearer ${ASANA_PAT}`, 'accept': 'application/json' }
        });
        const taskData = await response.json();
        
        if (!taskData || !taskData.data) return;
        const task = taskData.data;

        // 2. Look for the Application Number custom field
        const appNumberField = task.custom_fields.find(cf => cf.gid === CF_APP_NUMBER);
        
        if (appNumberField && appNumberField.text_value) {
            const applicationNumber = appNumberField.text_value;
            
            // 3. Fetch data from USPTO
            const usptoData = await fetchUSPTOData(applicationNumber);
            
            // 4. Update the task in Asana if data was found
            if (usptoData) {
                await updateAsanaTask(taskGid, usptoData);
            } else {
                console.log(`No data found for application number ${applicationNumber}`);
            }
        }
    } catch (error) {
        console.error("Error processing task event:", error);
    }
}

/**
 * Webhook Endpoint - Receives POST requests from Asana
 */
app.post('/webhook', async (req, res) => {
    // 1. Handle Asana Webhook Handshake
    if (req.headers['x-hook-secret']) {
        console.log("Received Asana Webhook Handshake");
        webhookSecret = req.headers['x-hook-secret'];
        // Asana requires the exact secret echoed back in the headers
        res.setHeader('X-Hook-Secret', webhookSecret);
        return res.sendStatus(200);
    }

    // 2. Verify Webhook Signature for security
    const signature = req.headers['x-hook-signature'];
    if (signature && webhookSecret) {
        const hmac = crypto.createHmac('sha256', webhookSecret);
        hmac.update(req.rawBody);
        const computedSignature = hmac.digest('hex');
        
        if (signature !== computedSignature) {
            console.error("Unauthorized: Signature mismatch. Ignored payload.");
            return res.sendStatus(401);
        }
    }

    // 3. Immediately respond with 200 OK so Asana knows we received it 
    // (If you don't respond within 10s, Asana will retry/disable the webhook)
    res.sendStatus(200);

    // 4. Process the events asynchronously
    const events = req.body.events || [];
    for (const event of events) {
        // Filter for when a task's custom fields are changed
        if (
            event.resource.resource_type === 'task' && 
            event.action === 'changed' &&
            event.change && event.change.field === 'custom_fields'
        ) {
            await processTaskEvent(event.resource.gid);
        }
    }
});

app.listen(PORT, () => {
    console.log(`Server is running on port ${PORT}`);
});
